xset s noblank # don't blank the video device

unclutter &    # hide X mouse cursor unless mouse activated
./.dotnet/dotnet ./kioskapp/$STARTUP --contentroot $(realpath ./kioskapp) &
chromium --display=:0 --kiosk --incognito --window-position=0,0 chrome://version/
